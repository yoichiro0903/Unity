;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; プロローグイベント
;;
;;	スクリプト読み込み直後に自動的に実行されるイベントです。
;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
Begin
	prologue
	set			Hero		vs_killer_bee	1
	set			Hero		ev_00_step		5
End

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; 勇者と村人１のイベント
;;
;; 薬を持っていないとき
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
Begin
	target		Hero
	target		Folk1
	condition	Hero		has_potion	0

	dialog		Folk1		これをあげます
	dialog		Hero		ありがとう～
	Text					薬を手に入れた！

	set			Hero		has_potion	1
End

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; 勇者と村人１のイベント
;;
;; 薬を１個持っているとき
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
Begin
	target		Hero
	target		Folk1
	condition	Hero		has_potion	1

	dialog		Folk1		もういっこあった
	dialog		Hero		やった！
	Text					勇者は小おどりした！

	set			Hero		has_potion	2
End

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; 勇者と村人１のイベント
;;
;; 薬を２個持っているとき
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
Begin
	target		Hero
	target		Folk1
	condition	Hero		has_potion	2

	dialog		Folk1		もうないよ
	dialog		Hero		がーん！
	Text					勇者は涙した……
End
